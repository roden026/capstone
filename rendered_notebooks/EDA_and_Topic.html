# calculate SC for K=2 through K=40 
# 40 is probably excessive but as you can see below, none of the results look encouraging so I wanted to stretch the number of groups and see how it went.
k_range = range(2, 41,1)
scores = []
for k in k_range:
    km = KMeans(n_clusters=k, random_state=1, n_jobs=-1, algorithm="auto", n_init=5, )
    km.fit(X)
    scores.append(metrics.silhouette_score(X, km.labels_))
    print("{} completed".format((k)))
